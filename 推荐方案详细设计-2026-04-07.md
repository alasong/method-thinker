## 二、Layer 0：快速过滤层

### 2.1 设计目标

```
目标：用最小成本快速识别明显问题，避免后续资源浪费

处理内容：
├── 语法检查：JSON/YAML格式是否正确
├── 字段完整性：必要字段是否齐全
├── 值域检查：分数、难度等是否在合理范围
└── 去重检查：是否与现有方法重复
```

### 2.2 实现代码

```python
# src/validation/layer0_fast_filter.py

from dataclasses import dataclass
from typing import List, Dict, Optional
import re
import json

@dataclass
class ValidationResult:
    """验证结果"""
    passed: bool
    layer: int
    confidence: float
    issues: List[str]
    details: Dict


class Layer0FastFilter:
    """Layer 0: 快速过滤器"""
    
    # 必要字段定义
    REQUIRED_FIELDS = [
        'method_id', 
        'name', 
        'description', 
        'applicability',
        'template'
    ]
    
    # 值域约束
    VALUE_CONSTRAINTS = {
        'difficulty': (1, 5),
        'frequency': (0, 1),
        'success_rate': (0, 1)
    }
    
    def __init__(self, existing_kb):
        self.existing_kb = existing_kb
        self.existing_names = {m.name.lower() for m in existing_kb.methods.values()}
        self.existing_ids = set(existing_kb.methods.keys())
    
    def validate(self, method: Dict) -> ValidationResult:
        """快速验证"""
        
        issues = []
        
        # 1. 字段完整性检查
        missing_fields = self._check_required_fields(method)
        if missing_fields:
            issues.append(f"缺少必要字段: {missing_fields}")
        
        # 2. 字段格式检查
        format_issues = self._check_field_formats(method)
        issues.extend(format_issues)
        
        # 3. 值域检查
        value_issues = self._check_value_constraints(method)
        issues.extend(value_issues)
        
        # 4. 去重检查
        duplicate_issues = self._check_duplicates(method)
        issues.extend(duplicate_issues)
        
        # 5. 描述质量检查
        quality_issues = self._check_description_quality(method)
        issues.extend(quality_issues)
        
        # 判断是否通过
        passed = len(issues) == 0
        confidence = 1.0 if passed else 0.0
        
        return ValidationResult(
            passed=passed,
            layer=0,
            confidence=confidence,
            issues=issues,
            details={'fast_filter': 'passed' if passed else 'failed'}
        )
    
    def _check_required_fields(self, method: Dict) -> List[str]:
        """检查必要字段"""
        return [f for f in self.REQUIRED_FIELDS if f not in method]
    
    def _check_field_formats(self, method: Dict) -> List[str]:
        """检查字段格式"""
        issues = []
        
        # method_id 格式检查
        if 'method_id' in method:
            if not re.match(r'^[A-Z]{3}_\d{3}$', method['method_id']):
                issues.append(f"method_id格式错误: {method['method_id']}")
        
        # name 长度检查
        if 'name' in method:
            if len(method['name']) < 2 or len(method['name']) > 50:
                issues.append(f"name长度异常: {len(method['name'])}")
        
        # description 长度检查
        if 'description' in method:
            if len(method['description']) < 20:
                issues.append("description过短，可能不够详细")
        
        # applicability 非空检查
        if 'applicability' in method:
            if not method['applicability'] or len(method['applicability']) == 0:
                issues.append("applicability为空")
        
        # template.steps 非空检查
        if 'template' in method:
            steps = method['template'].get('steps', [])
            if not steps or len(steps) < 2:
                issues.append("template.steps过少，方法步骤不完整")
        
        return issues
    
    def _check_value_constraints(self, method: Dict) -> List[str]:
        """检查值域约束"""
        issues = []
        
        for field, (min_val, max_val) in self.VALUE_CONSTRAINTS.items():
            if field in method:
                value = method[field]
                if not (min_val <= value <= max_val):
                    issues.append(f"{field}值{value}超出范围[{min_val}, {max_val}]")
        
        return issues
    
    def _check_duplicates(self, method: Dict) -> List[str]:
        """检查重复"""
        issues = []
        
        # ID重复检查
        if method.get('method_id') in self.existing_ids:
            issues.append(f"method_id重复: {method['method_id']}")
        
        # 名称相似度检查
        name = method.get('name', '').lower()
        for existing_name in self.existing_names:
            similarity = self._compute_similarity(name, existing_name)
            if similarity > 0.9:
                issues.append(f"方法名称与现有方法'{existing_name}'高度相似")
                break
        
        return issues
    
    def _check_description_quality(self, method: Dict) -> List[str]:
        """检查描述质量"""
        issues = []
        
        description = method.get('description', '')
        
        # 检查是否过于泛化
        vague_phrases = ['通用的方法', '一般的方法', '常规方法', '正常方法']
        if any(phrase in description for phrase in vague_phrases):
            issues.append("描述过于泛化，缺乏具体性")
        
        # 检查是否有实质性内容
        if len(description.split()) < 10:
            issues.append("描述内容过于简单")
        
        return issues
    
    @staticmethod
    def _compute_similarity(s1: str, s2: str) -> float:
        """计算字符串相似度（简化版Jaccard）"""
        if not s1 or not s2:
            return 0.0
        
        set1 = set(s1)
        set2 = set(s2)
        
        intersection = len(set1 & set2)
        union = len(set1 | set2)
        
        return intersection / union if union > 0 else 0.0
```

## 三、Layer 1：自我反思层

### 3.1 设计目标

```
目标：让模型自己检查自己，模拟人类自我纠错

核心理念：
├── 角色切换：解题者 → 批评者
├── 多角度反思：逻辑、适用性、完整性
├── 迭代改进：发现问题后尝试修正
└── 收敛判断：改进是否稳定
```

### 3.2 实现代码

```python
# src/validation/layer1_self_reflection.py

from typing import List, Dict
import json

class Layer1SelfReflection:
    """Layer 1: 自我反思验证"""
    
    def __init__(self, model, max_iterations: int = 3):
        self.model = model
        self.max_iterations = max_iterations
    
    def validate(self, method: Dict) -> ValidationResult:
        """自我反思验证"""
        
        # 记录反思历史
        reflection_history = []
        current_method = method
        
        for iteration in range(self.max_iterations):
            # 执行自我反思
            critique = self._self_critique(current_method)
            
            reflection_history.append({
                'iteration': iteration,
                'method': current_method,
                'critique': critique
            })
            
            # 如果通过自我批评，结束
            if critique['passed']:
                return ValidationResult(
                    passed=True,
                    layer=1,
                    confidence=critique['confidence'],
                    issues=[],
                    details={'reflections': reflection_history}
                )
            
            # 否则尝试改进
            if iteration < self.max_iterations - 1:
                improved = self._improve_method(current_method, critique)
                current_method = improved
        
        # 多次反思仍未通过
        return ValidationResult(
            passed=False,
            layer=1,
            confidence=0.0,
            issues=critique['issues'],
            details={'reflections': reflection_history}
        )
    
    def _self_critique(self, method: Dict) -> Dict:
        """自我批判"""
        
        prompt = f"""你是一位数学方法论专家，现在需要从批评的角度审视以下方法论：

方法名称：{method['name']}
方法描述：{method['description']}
适用条件：{method.get('applicability', [])}
执行步骤：{method.get('template', {}).get('steps', [])}

请从以下角度批判（注意：你要找问题，而不是说好话）：

1. 【逻辑一致性】
   - 各步骤之间是否逻辑连贯？
   - 是否有跳跃或遗漏的关键步骤？
   - 前后是否有矛盾？

2. 【适用范围】
   - 适用条件是否明确？还是过于模糊？
   - 是否遗漏了重要的特殊情况？
   - 是否有不适用但未说明的情况？

3. 【可执行性】
   - 步骤是否足够具体？能直接执行吗？
   - 是否需要未说明的前置知识或技巧？
   - 新手看了能理解怎么用吗？

4. 【完整性】
   - 是否有常见陷阱未提醒？
   - 是否有重要技巧未说明？
   - 是否有替代方案未提及？

输出JSON格式：
{{
    "passed": true/false,
    "confidence": 0.0-1.0,
    "issues": [
        {{"aspect": "逻辑一致性", "problem": "具体问题", "severity": "高/中/低"}},
        ...
    ],
    "suggestions": ["改进建议1", "改进建议2", ...]
}}

注意：置信度(confidence)表示你认为这个方法靠谱的程度。
如果找不到明显问题，passed=true，confidence=0.8-1.0
如果有严重问题，passed=false，confidence=0.0-0.3
"""
        
        response = self.model.generate(prompt, temperature=0.3)
        
        try:
            critique = json.loads(response)
        except:
            # 解析失败，保守判断
            critique = {
                'passed': False,
                'confidence': 0.3,
                'issues': [{'aspect': '格式', 'problem': '自我反思解析失败', 'severity': '中'}],
                'suggestions': ['请重新生成']
            }
        
        return critique
    
    def _improve_method(self, method: Dict, critique: Dict) -> Dict:
        """根据批判改进方法"""
        
        prompt = f"""你是一位数学方法论专家，需要根据批评意见改进方法论。

原始方法论：
{json.dumps(method, ensure_ascii=False, indent=2)}

批评意见：
{json.dumps(critique, ensure_ascii=False, indent=2)}

请改进方法论，解决上述问题。输出改进后的完整方法论JSON。
注意：保持原有字段，只改进有问题的地方。
"""
        
        response = self.model.generate(prompt, temperature=0.5)
        
        try:
            improved = json.loads(response)
            # 保留原有的method_id
            improved['method_id'] = method.get('method_id', improved.get('method_id'))
            return improved
        except:
            return method  # 解析失败，返回原方法
    
    def multi_perspective_reflection(self, method: Dict) -> Dict:
        """多角度反思（增强版）"""
        
        perspectives = [
            {
                'role': '初学者',
                'focus': '这个描述我能理解吗？步骤够清晰吗？',
                'temperature': 0.7
            },
            {
                'role': '专家',
                'focus': '这个方法在数学上严谨吗？有理论漏洞吗？',
                'temperature': 0.2
            },
            {
                'role': '教师',
                'focus': '这个方法适合教学吗？学生会理解吗？',
                'temperature': 0.5
            }
        ]
        
        reflections = []
        
        for p in perspectives:
            prompt = f"""你是一位{p['role']}，请从你的角度评价以下方法论：

方法：{method['name']}
描述：{method['description']}

关注点：{p['focus']}

请给出你的评价和建议（100字以内）。
"""
            response = self.model.generate(prompt, temperature=p['temperature'])
            reflections.append({
                'perspective': p['role'],
                'feedback': response
            })
        
        # 综合判断
        positive_count = sum(1 for r in reflections if '好' in r['feedback'] or '清晰' in r['feedback'])
        
        return {
            'reflections': reflections,
            'overall_positive': positive_count >= 2,
            'confidence': positive_count / len(perspectives)
        }
```

```python
# src/validation/reflection_monitor.py

class ReflectionMonitor:
    """反思效果监控"""
    
    def __init__(self):
        self.history = []
    
    def log_reflection(self, method_id: str, iteration: int, 
                       before_confidence: float, after_confidence: float):
        """记录反思迭代"""
        self.history.append({
            'method_id': method_id,
            'iteration': iteration,
            'before': before_confidence,
            'after': after_confidence,
            'improvement': after_confidence - before_confidence
        })
    
    def analyze_trends(self) -> Dict:
        """分析反思趋势"""
        
        if not self.history:
            return {'status': 'no_data'}
        
        # 平均改进
        avg_improvement = sum(h['improvement'] for h in self.history) / len(self.history)
        
        # 改进率
        improvement_rate = sum(1 for h in self.history if h['improvement'] > 0) / len(self.history)
        
        # 收敛情况
        recent = self.history[-10:]
        recent_improvements = [h['improvement'] for h in recent]
        is_converging = all(abs(imp) < 0.05 for imp in recent_improvements[-3:])
        
        return {
            'total_reflections': len(self.history),
            'avg_improvement': avg_improvement,
            'improvement_rate': improvement_rate,
            'is_converging': is_converging,
            'status': 'healthy' if improvement_rate > 0.5 else 'needs_attention'
        }
```

## 四、Layer 2：多模型交叉验证层（核心）

### 4.1 设计目标

```
目标：用多个模型相互验证，打破单一模型偏见

核心机制：
├── 多样性：不同模型有不同的"世界观"
├── 互补性：各模型优势领域不同
├── 冗余性：多数投票提供容错
└── 客观性：外部模型的判断更客观
```

```python
from dataclasses import dataclass

            strength=['推理', '语言'],

            latency=2.0

class ModelSelector:
    """模型选择器"""
    
    def __init__(self, budget: float = 500.0):
        self.budget = budget
        self.registry = ModelConfig.MODEL_REGISTRY
    
    def select_models(self, method: Dict) -> List[ModelConfig]:
        """选择验证模型"""
        
        # 根据方法类别选择模型
        category = method.get('category', 'GENERAL')
        
        selection_rules = {
            'ALGEBRA': ['deepseek_v3', 'qwen_math_72b', 'gpt4o_mini'],
            'GEOMETRY': ['qwen_math_72b', 'deepseek_v3', 'claude_haiku'],
            'NUMBER_THEORY': ['qwen_math_72b', 'deepseek_v3', 'gpt4o_mini'],
            'COMBINATORICS': ['deepseek_v3', 'claude_haiku', 'gpt4o_mini'],
            'CODE': ['deepseek_v3', 'gpt4o_mini', 'claude_haiku'],
            'GENERAL': ['deepseek_v3', 'gpt4o_mini', 'claude_haiku']
        }
        
        model_keys = selection_rules.get(category, selection_rules['GENERAL'])
        
        # 检查预算
        total_cost = sum(self.registry[k].cost_per_call for k in model_keys)
        if total_cost > self.budget * 0.1:  # 单次验证不超过预算10%
            # 选择成本更低的组合
            model_keys = ['gpt4o_mini', 'claude_haiku']
        
        return [self.registry[k] for k in model_keys]
```

```python
import asyncio
from typing import List, Dict
from dataclasses import dataclass
import json

@dataclass
class ModelAssessment:
    """单个模型的评估"""
    model_name: str
    score: float  # 0-10
    confidence: float  # 0-1
    issues: List[str]
    reasoning: str


class Layer2MultiModelValidation:
    """Layer 2: 多模型交叉验证"""
    
    def __init__(self, model_clients: Dict, budget: float = 500.0):
        self.clients = model_clients
        self.selector = ModelSelector(budget)
        self.budget_tracker = BudgetTracker(budget)
    
    def validate(self, method: Dict) -> ValidationResult:
        """多模型验证"""
        
        # 选择验证模型
        selected_models = self.selector.select_models(method)
        
        # 并行调用各模型
        assessments = self._parallel_assess(method, selected_models)
        
        # 集成决策
        decision = self._ensemble_decision(assessments)
        
        return ValidationResult(
            passed=decision['passed'],
            layer=2,
            confidence=decision['confidence'],
            issues=decision['issues'],
            details={
                'assessments': [a.__dict__ for a in assessments],
                'decision': decision
            }
        )
    
    def _parallel_assess(self, method: Dict, 
                         models: List[ModelConfig]) -> List[ModelAssessment]:
        """并行评估"""
        
        # 构建评估提示
        prompt = self._build_assessment_prompt(method)
        
        # 并行调用
        tasks = []
        for model_config in models:
            task = self._call_model_async(model_config, prompt)
            tasks.append(task)
        
        # 等待所有结果
        results = asyncio.run(self._gather_results(tasks))
        
        # 解析结果
        assessments = []
        for model_config, result in zip(models, results):
            assessment = self._parse_assessment(model_config.name, result)
            assessments.append(assessment)
            
            # 记录成本
            self.budget_tracker.record_cost(model_config.cost_per_call)
        
        return assessments
    
    def _build_assessment_prompt(self, method: Dict) -> str:
        """构建评估提示"""
        
        return f"""你是一位数学方法论评审专家，请评估以下方法论的质量。

方法名称：{method['name']}
方法描述：{method['description']}
适用条件：{method.get('applicability', [])}
执行步骤：{method.get('template', {}).get('steps', [])}

请从以下维度评分（每项0-10分）：

1. 【正确性】方法原理是否正确？有没有数学错误？
2. 【完整性】步骤是否完整？有没有遗漏？
3. 【适用性】适用条件是否合理？是否覆盖主要情况？
4. 【清晰度】描述是否清晰？能否理解如何使用？
5. 【实用性】这个方法有实际价值吗？值得学习吗？

输出JSON格式：
{{
    "scores": {{
        "correctness": 分数,
        "completeness": 分数,
        "applicability": 分数,
        "clarity": 分数,
        "practicality": 分数
    }},
    "overall_score": 综合分数（0-10）,
    "confidence": 你的把握程度（0-1）,
    "issues": ["发现的问题1", "发现的问题2", ...],
    "reasoning": "你的评估理由（100字以内）"
}}

注意：
- 如果发现严重错误（数学错误、逻辑错误），overall_score应低于5
- 如果没有明显问题但也不出色，overall_score在6-7
- 如果方法优秀，overall_score在8-10
"""
    
    async def _call_model_async(self, model_config: ModelConfig, 
                                 prompt: str) -> str:
        """异步调用模型"""
        
        client = self.clients[model_config.name]
        
        try:
            response = await client.generate_async(
                prompt,
                temperature=0.3,
                max_tokens=500
            )
            return response
        except Exception as e:
            return json.dumps({
                'error': str(e),
                'overall_score': 5,
                'confidence': 0.3,
                'issues': [f'模型调用失败: {e}'],
                'reasoning': '模型调用失败，保守评估'
            })
    
    async def _gather_results(self, tasks: List) -> List:
        """收集异步结果"""
        return await asyncio.gather(*tasks, return_exceptions=True)
    
    def _parse_assessment(self, model_name: str, response: str) -> ModelAssessment:
        """解析评估结果"""
        
        try:
            data = json.loads(response)
            
            return ModelAssessment(
                model_name=model_name,
                score=data.get('overall_score', 5),
                confidence=data.get('confidence', 0.5),
                issues=data.get('issues', []),
                reasoning=data.get('reasoning', '')
            )
        except:
            return ModelAssessment(
                model_name=model_name,
                score=5,
                confidence=0.3,
                issues=['评估结果解析失败'],
                reasoning='解析失败'
            )
    
    def _ensemble_decision(self, assessments: List[ModelAssessment]) -> Dict:
        """集成决策"""
        
        # 方法1：加权平均
        weighted_score = sum(
            a.score * a.confidence 
            for a in assessments
        ) / sum(a.confidence for a in assessments)
        
        # 方法2：多数投票
        approve_count = sum(1 for a in assessments if a.score >= 7)
        total = len(assessments)
        approve_rate = approve_count / total
        
        # 方法3：否决机制
        # 如果有模型给出低于5分，需要特别关注
        veto_count = sum(1 for a in assessments if a.score < 5)
        
        # 综合判断
        if veto_count >= 2:
            # 两个或以上模型给出低分，否决
            passed = False
            confidence = 0.9
        elif approve_rate >= 0.6:
            # 多数认可，通过
            passed = True
            confidence = approve_rate
        else:
            # 意见分歧
            passed = False
            confidence = 0.5
        
        # 收集问题
        all_issues = []
        for a in assessments:
            all_issues.extend([f"[{a.model_name}] {issue}" for issue in a.issues])
        
        # 去重
        unique_issues = list(set(all_issues))
        
        return {
            'passed': passed,
            'confidence': confidence,
            'weighted_score': weighted_score,
            'approve_rate': approve_rate,
            'issues': unique_issues,
            'assessments': assessments
        }


class BudgetTracker:
    """预算追踪器"""
    
    def __init__(self, total_budget: float):
        self.total_budget = total_budget
        self.spent = 0.0
        self.history = []
    
    def record_cost(self, cost: float):
        """记录成本"""
        self.spent += cost
        self.history.append(cost)
    
    def get_remaining(self) -> float:
        """获取剩余预算"""
        return self.total_budget - self.spent
    
    def can_afford(self, estimated_cost: float) -> bool:
        """判断是否负担得起"""
        return self.spent + estimated_cost <= self.total_budget * 0.95
```

```python
class AdversarialValidation:
```

```python
# src/validation/layer3_test_driven.py

from typing import List, Dict
import numpy as np
from dataclasses import dataclass

@dataclass
class TestCase:
    """测试用例"""
    problem: str
    answer: str
    difficulty: int
    problem_type: str
    method_hint: str = None


@dataclass
class TestResult:
    """测试结果"""
    test_case: TestCase
    predicted_answer: str
    is_correct: bool
    execution_time: float
    steps_count: int


class Layer3TestDrivenValidation:
    """Layer 3: 测试驱动验证"""
    
    def __init__(self, model, test_dataset: List[TestCase], 
                 pass_threshold: float = 0.6):
        self.model = model
        self.test_dataset = test_dataset
        self.pass_threshold = pass_threshold
    
    def validate(self, method: Dict) -> ValidationResult:
        """测试驱动验证"""
        
        # 根据方法筛选相关测试
        relevant_tests = self._select_relevant_tests(method)
        
        # 执行测试
        test_results = self._execute_tests(method, relevant_tests)
        
        # 计算统计指标
        stats = self._compute_statistics(test_results)
        
        # 判断是否通过
        passed = stats['success_rate'] >= self.pass_threshold
        
        return ValidationResult(
            passed=passed,
            layer=3,
            confidence=stats['confidence'],
            issues=stats['issues'],
            details={
                'statistics': stats,
                'test_results': [r.__dict__ for r in test_results[:5]]  # 只保存前5个详情
            }
        )
    
    def _select_relevant_tests(self, method: Dict) -> List[TestCase]:
        """选择相关测试"""
        
        # 根据方法适用条件筛选
        applicability = method.get('applicability', [])
        
        relevant = []
        for test in self.test_dataset:
            # 检查题型匹配
            if self._matches_applicability(test, applicability):
                relevant.append(test)
        
        # 限制测试数量
        return relevant[:50]  # 最多50个测试
    
    def _execute_tests(self, method: Dict, tests: List[TestCase]) -> List[TestResult]:
        """执行测试"""
        
        results = []
        
        for test in tests:
            # 构建输入
            input_text = self._build_test_input(test, method)
            
            # 模型生成解答
            start_time = time.time()
            output = self.model.generate(input_text, temperature=0.6)
            execution_time = time.time() - start_time
            
            # 提取答案
            predicted = self._extract_answer(output)
            
            # 验证正确性
            is_correct = self._verify_answer(predicted, test.answer)
            
            # 统计步骤数
            steps_count = self._count_steps(output)
            
            results.append(TestResult(
                test_case=test,
                predicted_answer=predicted,
                is_correct=is_correct,
                execution_time=execution_time,
                steps_count=steps_count
            ))
        
        return results
    
    def _compute_statistics(self, results: List[TestResult]) -> Dict:
        """计算统计指标"""
        
        if not results:
            return {
                'success_rate': 0,
                'confidence': 0,
                'issues': ['无相关测试用例']
            }
        
        # 成功率
        success_rate = sum(r.is_correct for r in results) / len(results)
        
        # 按难度分组统计
        difficulty_stats = {}
        for r in results:
            d = r.test_case.difficulty
            if d not in difficulty_stats:
                difficulty_stats[d] = {'correct': 0, 'total': 0}
            difficulty_stats[d]['total'] += 1
            if r.is_correct:
                difficulty_stats[d]['correct'] += 1
        
        # 识别问题
        issues = []
        
        # 整体成功率过低
        if success_rate < 0.4:
            issues.append(f"成功率过低({success_rate:.1%})，方法可能无效")
        
        # 难度梯度异常
        if 1 in difficulty_stats and 5 in difficulty_stats:
            easy_rate = difficulty_stats[1]['correct'] / difficulty_stats[1]['total']
            hard_rate = difficulty_stats[5]['correct'] / difficulty_stats[5]['total']
            if easy_rate < 0.7:
                issues.append("简单题成功率低，方法基础应用有问题")
        
        # 计算置信度
        # 基于样本量和成功率的标准误
        n = len(results)
        se = np.sqrt(success_rate * (1 - success_rate) / n)
        confidence = max(0, 1 - 2 * se)  # 95%置信区间
        
        return {
            'success_rate': success_rate,
            'confidence': confidence,
            'total_tests': n,
            'difficulty_stats': difficulty_stats,
            'avg_execution_time': np.mean([r.execution_time for r in results]),
            'avg_steps': np.mean([r.steps_count for r in results]),
            'issues': issues
        }
    
    def _build_test_input(self, test: TestCase, method: Dict) -> str:
        """构建测试输入"""
        
        return f"""问题：{test.problem}

请使用【{method['name']}】解答。

方法说明：
{method['description']}

执行步骤：
{chr(10).join(f'{i+1}. {s}' for i, s in enumerate(method.get('template', {}).get('steps', [])))}

请按此方法解答问题。
"""
    
    def _extract_answer(self, output: str) -> str:
        """从输出提取答案"""
        
        # 尝试多种模式
        patterns = [
            r'最终答案[是为：:]\s*(.+)',
            r'答案[是为：:]\s*(.+)',
            r'因此[，,]?\s*(.+)',
            r'所以[，,]?\s*(.+)'
        ]
        
        for pattern in patterns:
            match = re.search(pattern, output)
            if match:
                return match.group(1).strip()
        
        # 返回最后一行
        return output.strip().split('\n')[-1]
    
    def _verify_answer(self, predicted: str, expected: str) -> bool:
        """验证答案"""
        
        # 简化：字符串匹配
        # 实际应该用更智能的匹配（数值等价、表达式等价等）
        
        predicted = predicted.strip().lower()
        expected = expected.strip().lower()
        
        # 直接匹配
        if predicted == expected:
            return True
        
        # 数值匹配
        try:
            pred_num = float(re.search(r'[-+]?\d*\.?\d+', predicted).group())
            exp_num = float(re.search(r'[-+]?\d*\.?\d+', expected).group())
            return abs(pred_num - exp_num) < 1e-6
        except:
            pass
        
        return False
    
    def _count_steps(self, output: str) -> int:
        """统计步骤数"""
        
        # 统计编号的步骤
        steps = re.findall(r'^\d+[\.、）]', output, re.MULTILINE)
        return len(steps)
```

```python
from typing import List, Dict
from dataclasses import dataclass

@dataclass
class LayerResult:
    """单层验证结果"""
    layer: int
    passed: bool
    confidence: float
    issues: List[str]
    weight: float


class EnsembleDecisionEngine:
    """集成决策引擎"""
    
    def __init__(self):
        # 各层权重
        self.layer_weights = {
            0: 0.05,  # 快速过滤权重低
            1: 0.15,  # 自我反思权重中低
            2: 0.40,  # 多模型验证权重高（核心）
            3: 0.40   # 测试验证权重高（核心）
        }
        
        # 否决阈值
        self.veto_threshold = 0.3
    
    def decide(self, layer_results: List[LayerResult]) -> ValidationResult:
        """综合决策"""
        
        # 1. 计算加权得分
        weighted_score = sum(
            r.confidence * self.layer_weights[r.layer] 
            for r in layer_results
        )
        
        # 2. 检查否决情况
        veto_layer = self._check_veto(layer_results)
        
        # 3. 检查一致通过
        all_passed = all(r.passed for r in layer_results)
        
        # 4. 检查一致失败
        core_layers_failed = any(
            not r.passed and r.layer in [2, 3]
            for r in layer_results
        )
        
        # 5. 综合判断
        if veto_layer is not None:
            # 有否决
            passed = False
            confidence = 0.9
            reason = f"Layer {veto_layer} 否决"
        
        elif all_passed:
            # 全部通过
            passed = True
            confidence = weighted_score
            reason = "所有层级验证通过"
        
        elif core_layers_failed:
            # 核心层失败
            passed = False
            confidence = 0.8
            reason = "核心验证层失败"
        
        else:
            # 有分歧
            passed = weighted_score > 0.5
            confidence = weighted_score
            reason = f"加权得分: {weighted_score:.2f}"
        
        # 6. 收集问题
        all_issues = []
        for r in layer_results:
            all_issues.extend([f"[L{r.layer}] {issue}" for issue in r.issues])
        
        return ValidationResult(
            passed=passed,
            layer=-1,  # 表示集成层
            confidence=confidence,
            issues=all_issues,
            details={
                'weighted_score': weighted_score,
                'layer_results': [r.__dict__ for r in layer_results],
                'reason': reason
            }

    def _check_veto(self, results: List[LayerResult]) -> int:
```
